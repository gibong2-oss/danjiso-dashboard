```js
/* 전역 기본값 */
Chart.defaults.font.family = 'Pretendard';
Chart.defaults.font.size   = 11;

const PALETTE = ['#9B8FE8','#5EC8E8','#FF7056','#FFC654','#F48FB1','#8B80D4','#52C41A'];

const barOpts = (stacked=false) => ({
  responsive: true, maintainAspectRatio: false,
  plugins: { legend: { position:'top' } },
  scales: {
    x: { stacked, grid:{ display:false } },
    y: { stacked, grid:{ color:'#F0F0F0' }, ticks:{ maxTicksLimit:5 } }
  }
});

/* 파이·도넛 옵션 (cutout: '60%' → 도넛, '0%' → 파이) */
const pieOpts = (cutout='0%') => ({
  responsive: true, maintainAspectRatio: false,
  cutout,
  plugins: { legend: { position:'right' } }
});

/* 라인 차트 옵션 */
const lineOpts = () => ({
  responsive: true, maintainAspectRatio: false,
  plugins: { legend: { position:'top' } },
  scales: {
    x: { grid:{ display:false } },
    y: { grid:{ color:'#F0F0F0' }, ticks:{ maxTicksLimit:5 } }
  }
});

/* 차트 재생성 헬퍼 */
const charts = {};
function renderChart(id, config) {
  if (charts[id]) charts[id].destroy();
  charts[id] = new Chart(document.getElementById(id), config);
}
```

## ⚠️ 대시보드 생성 시 반드시 지켜야 할 기술 규칙

### 1. Python 템플릿 문자열 내 JS 따옴표 이스케이프 금지
Python `'''...'''` 또는 `"..."` 템플릿 문자열 안에서 `\'`는 그냥 `'`로 처리된다.
따라서 JS 코드에서 `onclick="fn(\'key\')"` 처럼 쓰면 실제 출력은 `onclick="fn('key')"` → **JS 문자열 파괴**.

**금지 패턴:**
```python
HTML = '''
+' onchange="toggleCol(\\''+c.k+'\\',this.checked)">'
'''
```

**해결책 A (권장) — data 속성 + addEventListener로 분리:**
```javascript
// HTML 생성 시:
'<input type="checkbox" data-key="'+c.k+'" ...'
// 이벤트 바인딩 (별도):
el.querySelectorAll('[data-key]').forEach(function(inp){
  inp.addEventListener('change', function(){ toggleCol(this.dataset.key, this.checked); });
});
```

**해결책 B — 인라인 이벤트는 반드시 double-quote 속성 + single-quote 없이:**
```javascript
// 동적 키가 없는 경우에만 인라인 onClick 허용
'<button onclick="myFn()">클릭</button>'
```

**해결책 C — JS 파일을 Python 문자열 바깥에서 관리 (PARTS 방식):**
```python
JS_CODE = open('template.js').read()   # 별도 파일
HTML = PARTS_HTML + '<script>' + JS_CODE.replace('__DATA__', DJ) + '</script>'
```

### 2. Python 생성 HTML의 JS 구문 검증 필수

HTML 생성 직후 반드시 Node.js로 구문 검증:
```python
import subprocess, re
# JS 블록 추출
js = re.search(r'<script>(.*?)</script>', html, re.DOTALL).group(1)
with open('/tmp/_test.js','w') as f:
    f.write('var document={getElementById:()=>({appendChild:()=>{},innerHTML:"",textContent:"",classList:{add:()=>{},remove:()=>{}},querySelectorAll:()=>[],querySelector:()=>null,value:"",style:{}})};var window={addEventListener:()=>{}};var Chart={register:()=>{}};var ChartDataLabels={};\n')
    f.write(js)
# ⚠️ 반드시 --check 플래그 사용! node 단독 실행 시 Chart.js 등 브라우저 전용
# 라이브러리가 없어 TypeError가 발생하므로 구문 검사(parse only)로 충분.
result = subprocess.run(['node', '--check', '/tmp/_test.js'], capture_output=True, text=True)
if result.returncode != 0:
    print('JS SYNTAX ERROR:', result.stderr[:400])
    raise SystemExit(1)
print('JS syntax OK')
```
